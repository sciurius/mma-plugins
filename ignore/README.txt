This plugin ignores the input line.

    // MMA 'ignore' plugin example.

    Plugin Ignore

    // Normal
    @Ignore Whatever is here

    // Track
    Bass @Ignore Whatever is here

    // Block
    Begin @Ignore
       1 the quick brown fox
       2 jumps over the
	 lazy dog
    End
